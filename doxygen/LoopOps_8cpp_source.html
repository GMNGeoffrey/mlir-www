57 static LogicalResult verify(ForOp op) {
58  if (auto cst = dyn_cast_or_null<ConstantIndexOp>(op.step().getDefiningOp()))
59  if (cst.getValue() <= 0)
60  return op.emitOpError("constant step operand must be positive");
61 
62  // Check that the body defines as single block argument for the induction
63  // variable.
64  auto *body = op.getBody();
65  if (!body->getArgument(0).getType().isIndex())
66  return op.emitOpError(
67  "expected body first argument to be an index argument for "
68  "the induction variable");
69 
70  auto opNumResults = op.getNumResults();
71  if (opNumResults == 0)
72  return success();
73  // If ForOp defines values, check that the number and types of
74  // the defined values match ForOp initial iter operands and backedge
75  // basic block arguments.
76  if (op.getNumIterOperands() != opNumResults)
77  return op.emitOpError(
78  "mismatch in number of loop-carried values and defined values");
79  if (op.getNumRegionIterArgs() != opNumResults)
80  return op.emitOpError(
81  "mismatch in number of basic block args and defined values");
82  auto iterOperands = op.getIterOperands();
83  auto iterArgs = op.getRegionIterArgs();
84  auto opResults = op.getResults();
85  unsigned i = 0;
86  for (auto e : llvm::zip(iterOperands, iterArgs, opResults)) {
87  if (std::get<0>(e).getType() != std::get<2>(e).getType())
88  return op.emitOpError() << "types mismatch between " << i
89  << "th iter operand and defined value";
90  if (std::get<1>(e).getType() != std::get<2>(e).getType())
91  return op.emitOpError() << "types mismatch between " << i
92  << "th iter region arg and defined value";
93 
94  i++;
95  }
96  return success();
97 }

122 static ParseResult parseForOp(OpAsmParser &parser, OperationState &result) {
123  auto &builder = parser.getBuilder();
124  OpAsmParser::OperandType inductionVariable, lb, ub, step;
125  // Parse the induction variable followed by '='.
126  if (parser.parseRegionArgument(inductionVariable) || parser.parseEqual())
127  return failure();
128 
129  // Parse loop bounds.
130  Type indexType = builder.getIndexType();
131  if (parser.parseOperand(lb) ||
132  parser.resolveOperand(lb, indexType, result.operands) ||
133  parser.parseKeyword("to") || parser.parseOperand(ub) ||
134  parser.resolveOperand(ub, indexType, result.operands) ||
135  parser.parseKeyword("step") || parser.parseOperand(step) ||
136  parser.resolveOperand(step, indexType, result.operands))
137  return failure();
138 
139  // Parse the optional initial iteration arguments.
140  SmallVector<OpAsmParser::OperandType, 4> regionArgs, operands;
141  SmallVector<Type, 4> argTypes;
142  regionArgs.push_back(inductionVariable);
143 
144  if (succeeded(parser.parseOptionalKeyword("iter_args"))) {
145  // Parse assignment list and results type list.
146  if (parser.parseAssignmentList(regionArgs, operands) ||
147  parser.parseArrowTypeList(result.types))
148  return failure();
149  // Resolve input operands.
150  for (auto operand_type : llvm::zip(operands, result.types))
151  if (parser.resolveOperand(std::get<0>(operand_type),
152  std::get<1>(operand_type), result.operands))
153  return failure();
154  }
155  // Induction variable.
156  argTypes.push_back(indexType);
157  // Loop carried variables
158  argTypes.append(result.types.begin(), result.types.end());
159  // Parse the body region.
160  Region *body = result.addRegion();
161  if (regionArgs.size() != argTypes.size())
162  return parser.emitError(
163  parser.getNameLoc(),
164  "mismatch in number of loop-carried values and defined values");
165 
166  if (parser.parseRegion(*body, regionArgs, argTypes))
167  return failure();
168 
169  ForOp::ensureTerminator(*body, builder, result.location);
170 
171  // Parse the optional attribute list.
172  if (parser.parseOptionalAttrDict(result.attributes))
173  return failure();
174 
175  return success();
176 }

211 /// Given the region at `index`, or the parent operation if `index` is None,
212 /// return the successor regions. These are the regions that may be selected
213 /// during the flow of control. `operands` is a set of optional attributes that
214 /// correspond to a constant value for each operand, or null if that operand is
215 /// not a constant.
216 void ForOp::getSuccessorRegions(Optional<unsigned> index,
217  ArrayRef<Attribute> operands,
218  SmallVectorImpl<RegionSuccessor> &regions) {
219  // If the predecessor is the ForOp, branch into the body using the iterator
220  // arguments.
221  if (!index.hasValue()) {
222  regions.push_back(RegionSuccessor(&getLoopBody(), getRegionIterArgs()));
223  return;
224  }
225 
226  // Otherwise, the loop may branch back to itself or the parent operation.
227  assert(index.getValue() == 0 && "expected loop region");
228  regions.push_back(RegionSuccessor(&getLoopBody(), getRegionIterArgs()));
229  regions.push_back(RegionSuccessor(getResults()));
230 }

242  TypeRange resultTypes, Value cond, bool withElseRegion) {
243  result.addOperands(cond);
244  result.addTypes(resultTypes);
245 
246  Region *thenRegion = result.addRegion();
247  thenRegion->push_back(new Block());
248  if (resultTypes.empty())
249  IfOp::ensureTerminator(*thenRegion, builder, result.location);

276 static ParseResult parseIfOp(OpAsmParser &parser, OperationState &result) {
277  // Create the regions for 'then'.
278  result.regions.reserve(2);
279  Region *thenRegion = result.addRegion();
280  Region *elseRegion = result.addRegion();
281 
282  auto &builder = parser.getBuilder();
283  OpAsmParser::OperandType cond;
284  Type i1Type = builder.getIntegerType(1);
285  if (parser.parseOperand(cond) ||
286  parser.resolveOperand(cond, i1Type, result.operands))
287  return failure();
288  // Parse optional results type list.
289  if (parser.parseOptionalArrowTypeList(result.types))
290  return failure();
291  // Parse the 'then' region.
292  if (parser.parseRegion(*thenRegion, /*arguments=*/{}, /*argTypes=*/{}))
293  return failure();
294  IfOp::ensureTerminator(*thenRegion, parser.getBuilder(), result.location);
295 
296  // If we find an 'else' keyword then parse the 'else' region.
297  if (!parser.parseOptionalKeyword("else")) {
298  if (parser.parseRegion(*elseRegion, /*arguments=*/{}, /*argTypes=*/{}))
299  return failure();
300  IfOp::ensureTerminator(*elseRegion, parser.getBuilder(), result.location);
301  }
302 
303  // Parse the optional attribute list.
304  if (parser.parseOptionalAttrDict(result.attributes))
305  return failure();
306  return success();
307 }

334 /// Given the region at `index`, or the parent operation if `index` is None,
335 /// return the successor regions. These are the regions that may be selected
336 /// during the flow of control. `operands` is a set of optional attributes that
337 /// correspond to a constant value for each operand, or null if that operand is
338 /// not a constant.
339 void IfOp::getSuccessorRegions(Optional<unsigned> index,
340  ArrayRef<Attribute> operands,
341  SmallVectorImpl<RegionSuccessor> &regions) {
342  // The `then` and the `else` region branch back to the parent operation.
343  if (index.hasValue()) {
344  regions.push_back(RegionSuccessor(getResults()));
345  return;
346  }
347 
348  // Don't consider the else region if it is empty.
349  Region *elseRegion = &this->elseRegion();
350  if (elseRegion->empty())
351  elseRegion = nullptr;
352 
353  // Otherwise, the successor is dependent on the condition.
354  bool condition;
355  if (auto condAttr = operands.front().dyn_cast_or_null<IntegerAttr>()) {
356  condition = condAttr.getValue().isOneValue();
357  } else if (auto condAttr = operands.front().dyn_cast_or_null<BoolAttr>()) {
358  condition = condAttr.getValue();
359  } else {
360  // If the condition isn't constant, both regions may be executed.
361  regions.push_back(RegionSuccessor(&thenRegion()));
362  regions.push_back(RegionSuccessor(elseRegion));
363  return;
364  }
365 
366  // Add the successor regions using the condition.
367  regions.push_back(RegionSuccessor(condition ? &thenRegion() : elseRegion));
368 }

395 static LogicalResult verify(ParallelOp op) {
396  // Check that there is at least one value in lowerBound, upperBound and step.
397  // It is sufficient to test only step, because it is ensured already that the
398  // number of elements in lowerBound, upperBound and step are the same.
399  Operation::operand_range stepValues = op.step();
400  if (stepValues.empty())
401  return op.emitOpError(
402  "needs at least one tuple element for lowerBound, upperBound and step");
403 
404  // Check whether all constant step values are positive.
405  for (Value stepValue : stepValues)
406  if (auto cst = dyn_cast_or_null<ConstantIndexOp>(stepValue.getDefiningOp()))
407  if (cst.getValue() <= 0)
408  return op.emitOpError("constant step operand must be positive");
409 
410  // Check that the body defines the same number of block arguments as the
411  // number of tuple elements in step.
412  Block *body = op.getBody();
413  if (body->getNumArguments() != stepValues.size())
414  return op.emitOpError()
415  << "expects the same number of induction variables: "
416  << body->getNumArguments()
417  << " as bound and step values: " << stepValues.size();
418  for (auto arg : body->getArguments())
419  if (!arg.getType().isIndex())
420  return op.emitOpError(
421  "expects arguments for the induction variable to be of index type");
422 
423  // Check that the number of results is the same as the number of ReduceOps.
424  SmallVector<ReduceOp, 4> reductions(body->getOps<ReduceOp>());
425  auto resultsSize = op.results().size();
426  auto reductionsSize = reductions.size();
427  auto initValsSize = op.initVals().size();
428  if (resultsSize != reductionsSize)
429  return op.emitOpError()
430  << "expects number of results: " << resultsSize
431  << " to be the same as number of reductions: " << reductionsSize;
432  if (resultsSize != initValsSize)
433  return op.emitOpError()
434  << "expects number of results: " << resultsSize
435  << " to be the same as number of initial values: " << initValsSize;
436 
437  // Check that the types of the results and reductions are the same.
438  for (auto resultAndReduce : llvm::zip(op.results(), reductions)) {
439  auto resultType = std::get<0>(resultAndReduce).getType();
440  auto reduceOp = std::get<1>(resultAndReduce);
441  auto reduceType = reduceOp.operand().getType();
442  if (resultType != reduceType)
443  return reduceOp.emitOpError()
444  << "expects type of reduce: " << reduceType
445  << " to be the same as result type: " << resultType;
446  }
447  return success();
448 }

450 static ParseResult parseParallelOp(OpAsmParser &parser,
451  OperationState &result) {
452  auto &builder = parser.getBuilder();
453  // Parse an opening `(` followed by induction variables followed by `)`
454  SmallVector<OpAsmParser::OperandType, 4> ivs;
455  if (parser.parseRegionArgumentList(ivs, /*requiredOperandCount=*/-1,
456  OpAsmParser::Delimiter::Paren))
457  return failure();
458 
459  // Parse loop bounds.
460  SmallVector<OpAsmParser::OperandType, 4> lower;
461  if (parser.parseEqual() ||
462  parser.parseOperandList(lower, ivs.size(),
463  OpAsmParser::Delimiter::Paren) ||
464  parser.resolveOperands(lower, builder.getIndexType(), result.operands))
465  return failure();
466 
467  SmallVector<OpAsmParser::OperandType, 4> upper;
468  if (parser.parseKeyword("to") ||
469  parser.parseOperandList(upper, ivs.size(),
470  OpAsmParser::Delimiter::Paren) ||
471  parser.resolveOperands(upper, builder.getIndexType(), result.operands))
472  return failure();
473 
474  // Parse step values.
475  SmallVector<OpAsmParser::OperandType, 4> steps;
476  if (parser.parseKeyword("step") ||
477  parser.parseOperandList(steps, ivs.size(),
478  OpAsmParser::Delimiter::Paren) ||
479  parser.resolveOperands(steps, builder.getIndexType(), result.operands))
480  return failure();
481 
482  // Parse init values.
483  SmallVector<OpAsmParser::OperandType, 4> initVals;
484  if (succeeded(parser.parseOptionalKeyword("init"))) {
485  if (parser.parseOperandList(initVals, /*requiredOperandCount=*/-1,
486  OpAsmParser::Delimiter::Paren))
487  return failure();
488  }
489 
490  // Parse optional results in case there is a reduce.
491  if (parser.parseOptionalArrowTypeList(result.types))
492  return failure();
493 
494  // Now parse the body.
495  Region *body = result.addRegion();
496  SmallVector<Type, 4> types(ivs.size(), builder.getIndexType());
497  if (parser.parseRegion(*body, ivs, types))
498  return failure();
499 
500  // Set `operand_segment_sizes` attribute.
501  result.addAttribute(
502  ParallelOp::getOperandSegmentSizeAttr(),
503  builder.getI32VectorAttr({static_cast<int32_t>(lower.size()),
504  static_cast<int32_t>(upper.size()),
505  static_cast<int32_t>(steps.size()),
506  static_cast<int32_t>(initVals.size())}));
507 
508  // Parse attributes.
509  if (parser.parseOptionalAttrDict(result.attributes))
510  return failure();
511 
512  if (!initVals.empty())
513  parser.resolveOperands(initVals, result.types, parser.getNameLoc(),
514  result.operands);
515  // Add a terminator if none was parsed.
516  ForOp::ensureTerminator(*body, builder, result.location);
517 
518  return success();
519 }

590 static ParseResult parseReduceOp(OpAsmParser &parser, OperationState &result) {
591  // Parse an opening `(` followed by the reduced value followed by `)`
592  OpAsmParser::OperandType operand;
593  if (parser.parseLParen() || parser.parseOperand(operand) ||
594  parser.parseRParen())
595  return failure();
596 
597  Type resultType;
598  // Parse the type of the operand (and also what reduce computes on).
599  if (parser.parseColonType(resultType) ||
600  parser.resolveOperand(operand, resultType, result.operands))
601  return failure();
602 
603  // Now parse the body.
604  Region *body = result.addRegion();
605  if (parser.parseRegion(*body, /*arguments=*/{}, /*argTypes=*/{}))
606  return failure();
607 
608  return success();
609 }

621 static LogicalResult verify(ReduceReturnOp op) {
622  // The type of the return value should be the same type as the type of the
623  // operand of the enclosing ReduceOp.
624  auto reduceOp = cast<ReduceOp>(op.getParentOp());
625  Type reduceType = reduceOp.operand().getType();
626  if (reduceType != op.result().getType())
627  return op.emitOpError() << "needs to have type " << reduceType
628  << " (the type of the enclosing ReduceOp)";
629  return success();
630 }

632 //===----------------------------------------------------------------------===//
633 // YieldOp
634 //===----------------------------------------------------------------------===//
635 static LogicalResult verify(YieldOp op) {
636  auto parentOp = op.getParentOp();
637  auto results = parentOp->getResults();
638  auto operands = op.getOperands();
639 
640  if (isa<IfOp>(parentOp) || isa<ForOp>(parentOp)) {
641  if (parentOp->getNumResults() != op.getNumOperands())
642  return op.emitOpError() << "parent of yield must have same number of "
643  "results as the yield operands";
644  for (auto e : llvm::zip(results, operands)) {
645  if (std::get<0>(e).getType() != std::get<1>(e).getType())
646  return op.emitOpError()
647  << "types mismatch between yield op and its parent";
648  }
649  } else if (isa<ParallelOp>(parentOp)) {
650  if (op.getNumOperands() != 0)
651  return op.emitOpError()
652  << "yield inside loop.parallel is not allowed to have operands";
653  } else {
654  return op.emitOpError()
655  << "yield only terminates If, For or Parallel regions";
656  }
657 
658  return success();
659 }

662  SmallVector<OpAsmParser::OperandType, 4> operands;

669  parser.resolveOperands(operands, types, loc, result.operands))
670  return failure();
671  return success();
