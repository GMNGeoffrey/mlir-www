28 /// Performs standalone tiling of a single LinalgOp by `tileSizes`.
29 /// and permute the loop nest according to `interchangeVector`
30 /// The permutation is expressed as a list of integers that specify
31 /// the new ordering of the loop nest. The length of `interchangeVector`
32 /// must be equal to the length of `tileSizes`.
33 /// An empty vector is interpreted as the identity permutation and the
34 /// transformation returns early.
35 ///
36 /// When non-null, the optional pointer `folder` is used to call into the
37 /// `createAndFold` builder method. If `folder` is null, the regular `create`
38 /// method is called.
39 ///
40 /// Returns a struct containing the tiled loops in the specified order
41 /// and the cloned op if successful, llvm::None otherwise.
42 ///
43 /// E.g. the permutation `(i,j,k) -> (j,k,i)` is expressed by
44 /// `interchangeVector = [1,2,0]`. All values in `interchangeVector` must be
45 /// integers, in the range 0..`tileSizes.size()` without duplications
46 /// (i.e. `[1,1,2]` is an invalid permutation).
47 Optional<TiledLinalgOp> tileLinalgOp(OpBuilder &b, LinalgOp op,
48  ArrayRef<Value> tileSizes,
49  ArrayRef<unsigned> interchangeVector = {},
50  OperationFolder *folder = nullptr);

81 /// `b`. Promotion occurs in 3 steps:

318 ///
319 /// Linalg lowering patterns.
320 ///
321 /// Apply the `linalgLowerOpToLoops` transformation as a pattern.
322 /// `marker` controls LinalgTransformMarker matching and update when specified.
323 /// See `linalgLowerOpToLoops` for more details.
324 enum class LinalgLoweringType {
325  LibraryCall = 0,
326  Loops = 1,
327  AffineLoops = 2,
328  ParallelLoops = 3
329 };
330 template <typename OpTy>
331 struct LinalgLoweringPattern : public RewritePattern {
332  LinalgLoweringPattern(MLIRContext *context, LinalgLoweringType loweringType,
333  LinalgMarker marker = LinalgMarker(),
334  PatternBenefit benefit = 1)
335  : RewritePattern(OpTy::getOperationName(), {}, benefit, context),
336  marker(marker), loweringType(loweringType) {}
337  // TODO: Move implementation to .cpp once named ops are auto-generated.
338  LogicalResult matchAndRewrite(Operation *op,
339  PatternRewriter &rewriter) const override {
340  LinalgOp linalgOp = dyn_cast<LinalgOp>(op);
341  if (!linalgOp)
342  return failure();
343  if (failed(marker.checkAndNotify(rewriter, linalgOp)))
344  return failure();
345  if (failed(promoteSubviewsLinalgOpPrecondition(op)))
346  return failure();
347 
348  if (loweringType == LinalgLoweringType::LibraryCall) {
349  // TODO: Move lowering to library calls here.
350  return failure();
351  } else if (loweringType == LinalgLoweringType::Loops) {
352  if (failed(linalgOpToLoops<OpTy>(rewriter, op)))
353  return failure();
354  } else if (loweringType == LinalgLoweringType::AffineLoops) {
355  if (failed(linalgOpToAffineLoops<OpTy>(rewriter, op)))
356  return failure();
357  } else if (failed(linalgOpToParallelLoops<OpTy>(rewriter, op))) {
358  return failure();
359  }
360  rewriter.eraseOp(op);
361  return success();
362  }
363 
364 private:
365  /// LinalgTransformMarker handles special attribute manipulations.
366  LinalgMarker marker;
367  /// Controls whether the pattern lowers to library calls, loop.for, affine.for
368  /// or loop.parallel.
369  LinalgLoweringType loweringType;
370 };
