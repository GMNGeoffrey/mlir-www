9 // This file implements logic for computing correct alloc and dealloc positions.
10 // The main class is the BufferPlacementPass class that implements the
11 // underlying algorithm. In order to put allocations and deallocations at safe
12 // positions, it is significantly important to put them into the correct blocks.
13 // However, the liveness analysis does not pay attention to aliases, which can
14 // occur due to branches (and their associated block arguments) in general. For
15 // this purpose, BufferPlacement firstly finds all possible aliases for a single
16 // value (using the BufferPlacementAliasAnalysis class). Consider the following
17 // example:
18 //
19 // ^bb0(%arg0):
20 // cond_br %cond, ^bb1, ^bb2
21 // ^bb1:
22 // br ^exit(%arg0)
23 // ^bb2:
24 // %new_value = ...
25 // br ^exit(%new_value)
26 // ^exit(%arg1):
27 // return %arg1;
28 //
29 // Using liveness information on its own would cause us to place the allocs and
30 // deallocs in the wrong block. This is due to the fact that %new_value will not
31 // be liveOut of its block. Instead, we have to place the alloc for %new_value
32 // in bb0 and its associated dealloc in exit. Using the class
33 // BufferPlacementAliasAnalysis, we will find out that %new_value has a
34 // potential alias %arg1. In order to find the dealloc position we have to find
35 // all potential aliases, iterate over their uses and find the common
36 // post-dominator block. In this block we can safely be sure that %new_value
37 // will die and can use liveness information to determine the exact operation
38 // after which we have to insert the dealloc. Finding the alloc position is
39 // highly similar and non- obvious. Again, we have to consider all potential
40 // aliases and find the common dominator block to place the alloc.
41 //
42 // TODO:
43 // The current implementation does not support loops and the resulting code will
44 // be invalid with respect to program semantics. The only thing that is
45 // currently missing is a high-level loop analysis that allows us to move allocs
46 // and deallocs outside of the loop blocks.
47 //

64 /// A straight-forward alias analysis which ensures that all aliases of all
65 /// values will be determined. This is a requirement for the BufferPlacement
66 /// class since you need to determine safe positions to place alloc and
67 /// deallocs.
68 class BufferPlacementAliasAnalysis {
69 public:
70  using ValueSetT = SmallPtrSet<Value, 16>;
71 
72 public:
73  /// Constructs a new alias analysis using the op provided.
74  BufferPlacementAliasAnalysis(Operation *op) { build(op->getRegions()); }
75 
76  /// Finds all immediate and indirect aliases this value could potentially
77  /// have. Note that the resulting set will also contain the value provided as
78  /// it is an alias of itself.
79  ValueSetT resolve(Value value) const {
80  ValueSetT result;
81  resolveRecursive(value, result);
82  return result;
83  }
84 
85 private:
86  /// Recursively determines alias information for the given value. It stores
87  /// all newly found potential aliases in the given result set.
88  void resolveRecursive(Value value, ValueSetT &result) const {
89  if (!result.insert(value).second)
90  return;
91  auto it = aliases.find(value);
92  if (it == aliases.end())
93  return;
94  for (Value alias : it->second)
95  resolveRecursive(alias, result);
96  }
97 
98  /// This function constructs a mapping from values to its immediate aliases.
99  /// It iterates over all blocks, gets their predecessors, determines the
100  /// values that will be passed to the corresponding block arguments and
101  /// inserts them into the underlying map.
102  void build(MutableArrayRef<Region> regions) {
103  for (Region &region : regions) {
104  for (Block &block : region) {
105  // Iterate over all predecessor and get the mapped values to their
106  // corresponding block arguments values.
107  for (auto it = block.pred_begin(), e = block.pred_end(); it != e;
108  ++it) {
109  unsigned successorIndex = it.getSuccessorIndex();
110  // Get the terminator and the values that will be passed to our block.
111  auto branchInterface =
112  dyn_cast<BranchOpInterface>((*it)->getTerminator());
113  if (!branchInterface)
114  continue;
115  // Query the branch op interace to get the successor operands.
116  auto successorOperands =
117  branchInterface.getSuccessorOperands(successorIndex);
118  if (successorOperands.hasValue()) {
119  // Build the actual mapping of values to their immediate aliases.
120  for (auto argPair : llvm::zip(block.getArguments(),
121  successorOperands.getValue())) {
122  aliases[std::get<1>(argPair)].insert(std::get<0>(argPair));
123  }
124  }
125  }
126  }
127  }
128  }
129 
130  /// Maps values to all immediate aliases this value can have.
131  llvm::DenseMap<Value, ValueSetT> aliases;
132 };

166 // The main buffer placement analysis used to place allocs and deallocs.
167 class BufferPlacementAnalysis {
168 public:
169  using DeallocSetT = SmallPtrSet<Operation *, 2>;
170 
171 public:
172  BufferPlacementAnalysis(Operation *op)
173  : operation(op), liveness(op), dominators(op), postDominators(op),
174  aliases(op) {}
175 
176  /// Computes the actual positions to place allocs and deallocs for the given
177  /// value.
178  BufferPlacementPositions
179  computeAllocAndDeallocPositions(OpResult result) const {
180  if (result.use_empty())
181  return BufferPlacementPositions(result.getOwner(), result.getOwner());
182  // Get all possible aliases.
183  auto possibleValues = aliases.resolve(result);
184  return BufferPlacementPositions(getAllocPosition(result, possibleValues),
185  getDeallocPosition(result, possibleValues));
186  }
187 
188  /// Finds all associated dealloc nodes for the alloc nodes using alias
189  /// information.
190  DeallocSetT findAssociatedDeallocs(OpResult allocResult) const {
191  DeallocSetT result;
192  auto possibleValues = aliases.resolve(allocResult);
193  for (Value alias : possibleValues)
194  for (Operation *op : alias.getUsers()) {
195  // Check for an existing memory effect interface.
196  auto effectInstance = dyn_cast<MemoryEffectOpInterface>(op);
197  if (!effectInstance)
198  continue;
199  // Check whether the associated value will be freed using the current
200  // operation.
201  SmallVector<MemoryEffects::EffectInstance, 2> effects;
202  effectInstance.getEffectsOnValue(alias, effects);
203  if (llvm::any_of(effects, [=](MemoryEffects::EffectInstance &it) {
204  return isa<MemoryEffects::Free>(it.getEffect());
205  }))
206  result.insert(op);
207  }
208  return result;
209  }
210 
211  /// Dumps the buffer placement information to the given stream.
212  void print(raw_ostream &os) const {
213  os << "// ---- Buffer Placement -----\n";
214 
215  for (Region &region : operation->getRegions())
216  for (Block &block : region)
217  for (Operation &operation : block)
218  for (OpResult result : operation.getResults()) {
219  BufferPlacementPositions positions =
220  computeAllocAndDeallocPositions(result);
221  os << "Positions for ";
222  result.print(os);
223  os << "\n Alloc: ";
224  positions.getAllocPosition()->print(os);
225  os << "\n Dealloc: ";
226  positions.getDeallocPosition()->print(os);
227  os << "\n";
228  }
229  }
230 
231 private:
232  /// Finds a correct placement block to store alloc/dealloc node according to
233  /// the algorithm described at the top of the file. It supports dominator and
234  /// post-dominator analyses via template arguments.
235  template <typename DominatorT>
236  Block *
237  findPlacementBlock(OpResult result,
238  const BufferPlacementAliasAnalysis::ValueSetT &aliases,
239  const DominatorT &doms) const {
240  // Start with the current block the value is defined in.
241  Block *dom = result.getOwner()->getBlock();
242  // Iterate over all aliases and their uses to find a safe placement block
243  // according to the given dominator information.
244  for (Value alias : aliases)
245  for (Operation *user : alias.getUsers()) {
246  // Move upwards in the dominator tree to find an appropriate
247  // dominator block that takes the current use into account.
248  dom = doms.findNearestCommonDominator(dom, user->getBlock());
249  }
250  return dom;
251  }
252 
253  /// Finds a correct alloc position according to the algorithm described at
254  /// the top of the file.
255  Operation *getAllocPosition(
256  OpResult result,
257  const BufferPlacementAliasAnalysis::ValueSetT &aliases) const {
258  // Determine the actual block to place the alloc and get liveness
259  // information.
260  Block *placementBlock = findPlacementBlock(result, aliases, dominators);
261  const LivenessBlockInfo *livenessInfo =
262  liveness.getLiveness(placementBlock);
263 
264  // We have to ensure that the alloc will be before the first use of all
265  // aliases of the given value. We first assume that there are no uses in the
266  // placementBlock and that we can safely place the alloc before the
267  // terminator at the end of the block.
268  Operation *startOperation = placementBlock->getTerminator();
269  // Iterate over all aliases and ensure that the startOperation will point to
270  // the first operation of all potential aliases in the placementBlock.
271  for (Value alias : aliases) {
272  Operation *aliasStartOperation = livenessInfo->getStartOperation(alias);
273  // Check whether the aliasStartOperation lies in the desired block and
274  // whether it is before the current startOperation. If yes, this will be
275  // the new startOperation.
276  if (aliasStartOperation->getBlock() == placementBlock &&
277  aliasStartOperation->isBeforeInBlock(startOperation))
278  startOperation = aliasStartOperation;
279  }
280  // startOperation is the first operation before which we can safely store
281  // the alloc taking all potential aliases into account.
282  return startOperation;
283  }
284 
285  /// Finds a correct dealloc position according to the algorithm described at
286  /// the top of the file.
287  Operation *getDeallocPosition(
288  OpResult result,
289  const BufferPlacementAliasAnalysis::ValueSetT &aliases) const {
290  // Determine the actual block to place the dealloc and get liveness
291  // information.
292  Block *placementBlock = findPlacementBlock(result, aliases, postDominators);
293  const LivenessBlockInfo *livenessInfo =
294  liveness.getLiveness(placementBlock);
295 
296  // We have to ensure that the dealloc will be after the last use of all
297  // aliases of the given value. We first assume that there are no uses in the
298  // placementBlock and that we can safely place the dealloc at the beginning.
299  Operation *endOperation = &placementBlock->front();
300  // Iterate over all aliases and ensure that the endOperation will point to
301  // the last operation of all potential aliases in the placementBlock.
302  for (Value alias : aliases) {
303  Operation *aliasEndOperation =
304  livenessInfo->getEndOperation(alias, endOperation);
305  // Check whether the aliasEndOperation lies in the desired block and
306  // whether it is behind the current endOperation. If yes, this will be the
307  // new endOperation.
308  if (aliasEndOperation->getBlock() == placementBlock &&
309  endOperation->isBeforeInBlock(aliasEndOperation))
310  endOperation = aliasEndOperation;
311  }
312  // endOperation is the last operation behind which we can safely store the
313  // dealloc taking all potential aliases into account.
314  return endOperation;
315  }
316 
317  /// The operation this transformation was constructed from.
318  Operation *operation;
319 
320  /// The underlying liveness analysis to compute fine grained information about
321  /// alloc and dealloc positions.
322  Liveness liveness;
323 
324  /// The dominator analysis to place allocs in the appropriate blocks.
325  DominanceInfo dominators;
326 
327  /// The post dominator analysis to place deallocs in the appropriate blocks.
328  PostDominanceInfo postDominators;
329 
330  /// The internal alias analysis to ensure that allocs and deallocs take all
331  /// their potential aliases into account.
332  BufferPlacementAliasAnalysis aliases;
333 };

339 /// The actual buffer placement pass that moves alloc and dealloc nodes into
340 /// the right positions. It uses the algorithm described at the top of the file.
341 struct BufferPlacementPass
342  : mlir::PassWrapper<BufferPlacementPass, FunctionPass> {
343  void runOnFunction() override {
344  // Get required analysis information first.
345  auto &analysis = getAnalysis<BufferPlacementAnalysis>();
346 
347  // Compute an initial placement of all nodes.
348  llvm::SmallVector<std::pair<OpResult, BufferPlacementPositions>, 16>
349  placements;
350  getFunction().walk([&](MemoryEffectOpInterface op) {
351  // Try to find a single allocation result.
352  SmallVector<MemoryEffects::EffectInstance, 2> effects;
353  op.getEffects(effects);
354 
355  SmallVector<MemoryEffects::EffectInstance, 2> allocateResultEffects;
356  llvm::copy_if(effects, std::back_inserter(allocateResultEffects),
357  [=](MemoryEffects::EffectInstance &it) {
358  Value value = it.getValue();
359  return isa<MemoryEffects::Allocate>(it.getEffect()) &&
360  value && value.isa<OpResult>();
361  });
362  // If there is one result only, we will be able to move the allocation and
363  // (possibly existing) deallocation ops.
364  if (allocateResultEffects.size() == 1) {
365  // Insert allocation result.
366  auto allocResult = allocateResultEffects[0].getValue().cast<OpResult>();
367  placements.emplace_back(
368  allocResult, analysis.computeAllocAndDeallocPositions(allocResult));
369  }
370  });
371 
372  // Move alloc (and dealloc - if any) nodes into the right places and insert
373  // dealloc nodes if necessary.
374  for (auto &entry : placements) {
375  // Find already associated dealloc nodes.
376  OpResult alloc = entry.first;
377  auto deallocs = analysis.findAssociatedDeallocs(alloc);
378  if (deallocs.size() > 1) {
379  emitError(alloc.getLoc(),
380  "not supported number of associated dealloc operations");
381  return;
382  }
383 
384  // Move alloc node to the right place.
385  BufferPlacementPositions &positions = entry.second;
386  Operation *allocOperation = alloc.getOwner();
387  allocOperation->moveBefore(positions.getAllocPosition());
388 
389  // If there is an existing dealloc, move it to the right place.
390  Operation *nextOp = positions.getDeallocPosition()->getNextNode();
391  assert(nextOp && "Invalid Dealloc operation position");
392  if (deallocs.size()) {
393  (*deallocs.begin())->moveBefore(nextOp);
394  } else {
395  // If there is no dealloc node, insert one in the right place.
396  OpBuilder builder(nextOp);
397  builder.create<DeallocOp>(allocOperation->getLoc(), alloc);
398  }
399  }
400  };
401 };

423 // Performs the actual signature rewriting step.
424 LogicalResult FunctionAndBlockSignatureConverter::matchAndRewrite(
425  FuncOp funcOp, ArrayRef<Value> operands,
426  ConversionPatternRewriter &rewriter) const {
427  if (!converter) {
428  funcOp.emitError("The type converter has not been defined for "
429  "FunctionAndBlockSignatureConverter");
430  return failure();
431  }
432  // Converting shaped type arguments to memref type.
433  auto funcType = funcOp.getType();
434  TypeConverter::SignatureConversion conversion(funcType.getNumInputs());
435  for (auto argType : llvm::enumerate(funcType.getInputs()))
436  conversion.addInputs(argType.index(),
437  converter->convertType(argType.value()));
438  // Adding function results to the arguments of the converted function as
439  // memref type. The converted function will be a void function.
440  for (Type resType : funcType.getResults())
441  conversion.addInputs(converter->convertType((resType)));
442  rewriter.updateRootInPlace(funcOp, [&] {
443  funcOp.setType(
444  rewriter.getFunctionType(conversion.getConvertedTypes(), llvm::None));
445  rewriter.applySignatureConversion(&funcOp.getBody(), conversion);
446  });
447  return success();
448 }
