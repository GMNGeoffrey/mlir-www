34 ArrayRef<Extension> spirv::getImpliedExtensions(Version version) {
35  // Note: the following lists are from "Appendix A: Changes" of the spec.
36 
37 #define V_1_3_IMPLIED_EXTS \
38  Extension::SPV_KHR_shader_draw_parameters, Extension::SPV_KHR_16bit_storage, \
39  Extension::SPV_KHR_device_group, Extension::SPV_KHR_multiview, \
40  Extension::SPV_KHR_storage_buffer_storage_class, \
41  Extension::SPV_KHR_variable_pointers
42 
43 #define V_1_4_IMPLIED_EXTS \
44  Extension::SPV_KHR_no_integer_wrap_decoration, \
45  Extension::SPV_GOOGLE_decorate_string, \
46  Extension::SPV_GOOGLE_hlsl_functionality1, \
47  Extension::SPV_KHR_float_controls
48 
49 #define V_1_5_IMPLIED_EXTS \
50  Extension::SPV_KHR_8bit_storage, Extension::SPV_EXT_descriptor_indexing, \
51  Extension::SPV_EXT_shader_viewport_index_layer, \
52  Extension::SPV_EXT_physical_storage_buffer, \
53  Extension::SPV_KHR_physical_storage_buffer, \
54  Extension::SPV_KHR_vulkan_memory_model
55 
56  switch (version) {
57  default:
58  return {};
59  case Version::V_1_3: {
60  // The following manual ArrayRef constructor call is to satisfy GCC 5.
61  static const Extension exts[] = {V_1_3_IMPLIED_EXTS};
62  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
63  }
64  case Version::V_1_4: {
65  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS};
66  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
67  }
68  case Version::V_1_5: {
69  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS,
70  V_1_5_IMPLIED_EXTS};
71  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
72  }
73  }
74 
75 #undef V_1_5_IMPLIED_EXTS
76 #undef V_1_4_IMPLIED_EXTS
77 #undef V_1_3_IMPLIED_EXTS
78 }

83 SmallVector<Capability, 0>
84 spirv::getRecursiveImpliedCapabilities(Capability cap) {
85  ArrayRef<Capability> directCaps = getDirectImpliedCapabilities(cap);
86  llvm::SetVector<Capability, SmallVector<Capability, 0>> allCaps(
87  directCaps.begin(), directCaps.end());
88 
89  // TODO(antiagainst): This is insufficient; find a better way to handle this
90  // (e.g., using static lists) if this turns out to be a bottleneck.
91  for (unsigned i = 0; i < allCaps.size(); ++i)
92  for (Capability c : getDirectImpliedCapabilities(allCaps[i]))
93  allCaps.insert(c);
94 
95  return allCaps.takeVector();
96 }

603 void ScalarType::getExtensions(SPIRVType::ExtensionArrayRefVector &extensions,
604  Optional<StorageClass> storage) {
605  // 8- or 16-bit integer/floating-point numbers will require extra extensions
606  // to appear in interface storage classes. See SPV_KHR_16bit_storage and
607  // SPV_KHR_8bit_storage for more details.
608  if (!storage)
609  return;
610 
611  switch (*storage) {
612  case StorageClass::PushConstant:
613  case StorageClass::StorageBuffer:
614  case StorageClass::Uniform:
615  if (getIntOrFloatBitWidth() == 8) {
616  static const Extension exts[] = {Extension::SPV_KHR_8bit_storage};
617  ArrayRef<Extension> ref(exts, llvm::array_lengthof(exts));
618  extensions.push_back(ref);
619  }
620  LLVM_FALLTHROUGH;
621  case StorageClass::Input:
622  case StorageClass::Output:
623  if (getIntOrFloatBitWidth() == 16) {
624  static const Extension exts[] = {Extension::SPV_KHR_16bit_storage};
625  ArrayRef<Extension> ref(exts, llvm::array_lengthof(exts));
626  extensions.push_back(ref);
627  }
628  break;
629  default:
630  break;
631  }
632 }

634 void ScalarType::getCapabilities(
635  SPIRVType::CapabilityArrayRefVector &capabilities,
636  Optional<StorageClass> storage) {
637  unsigned bitwidth = getIntOrFloatBitWidth();
638 
639  // 8- or 16-bit integer/floating-point numbers will require extra capabilities
640  // to appear in interface storage classes. See SPV_KHR_16bit_storage and
641  // SPV_KHR_8bit_storage for more details.
642 
643 #define STORAGE_CASE(storage, cap8, cap16) \
644  case StorageClass::storage: { \
645  if (bitwidth == 8) { \
646  static const Capability caps[] = {Capability::cap8}; \
647  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
648  capabilities.push_back(ref); \
649  } else if (bitwidth == 16) { \
650  static const Capability caps[] = {Capability::cap16}; \
651  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
652  capabilities.push_back(ref); \
653  } \
654  } break
655 
656  if (storage) {
657  switch (*storage) {
658  STORAGE_CASE(PushConstant, StoragePushConstant8, StoragePushConstant16);
659  STORAGE_CASE(StorageBuffer, StorageBuffer8BitAccess,
660  StorageBuffer16BitAccess);
661  STORAGE_CASE(Uniform, UniformAndStorageBuffer8BitAccess,
662  StorageUniform16);
663  case StorageClass::Input:
664  case StorageClass::Output:
665  if (bitwidth == 16) {
666  static const Capability caps[] = {Capability::StorageInputOutput16};
667  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps));
668  capabilities.push_back(ref);
669  }
670  break;
671  default:
672  break;
673  }

679  // capabilities for special bitwidths.

682  case width: { \
683  static const Capability caps[] = {Capability::type##width}; \
684  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
685  capabilities.push_back(ref); \
686  } break
687 
688  if (auto intType = dyn_cast<IntegerType>()) {
689  switch (bitwidth) {
690  case 32:
691  case 1:
692  break;
693  WIDTH_CASE(Int, 8);
694  WIDTH_CASE(Int, 16);
695  WIDTH_CASE(Int, 64);
696  default:
697  llvm_unreachable("invalid bitwidth to getCapabilities");
698  }
699  } else {
700  assert(isa<FloatType>());
701  switch (bitwidth) {
702  case 32:
703  break;
704  WIDTH_CASE(Float, 16);
705  WIDTH_CASE(Float, 64);
706  default:
707  llvm_unreachable("invalid bitwidth to getCapabilities");
708  }
709  }

772  StructType::LayoutInfo const *layoutInfo, unsigned numMemberDecorations,
773  StructType::MemberDecorationInfo const *memberDecorationsInfo)
774  : TypeStorage(numMembers), memberTypes(memberTypes),
775  layoutInfo(layoutInfo), numMemberDecorations(numMemberDecorations),
776  memberDecorationsInfo(memberDecorationsInfo) {}
777 
778  using KeyTy = std::tuple<ArrayRef<Type>, ArrayRef<StructType::LayoutInfo>,
779  ArrayRef<StructType::MemberDecorationInfo>>;
780  bool operator==(const KeyTy &key) const {
781  return key ==
782  KeyTy(getMemberTypes(), getLayoutInfo(), getMemberDecorationsInfo());
783  }

785  static StructTypeStorage *construct(TypeStorageAllocator &allocator,
786  const KeyTy &key) {
787  ArrayRef<Type> keyTypes = std::get<0>(key);
788 
789  // Copy the member type and layout information into the bump pointer
790  const Type *typesList = nullptr;
791  if (!keyTypes.empty()) {
792  typesList = allocator.copyInto(keyTypes).data();
793  }
794 
795  const StructType::LayoutInfo *layoutInfoList = nullptr;
796  if (!std::get<1>(key).empty()) {
797  ArrayRef<StructType::LayoutInfo> keyLayoutInfo = std::get<1>(key);
798  assert(keyLayoutInfo.size() == keyTypes.size() &&
799  "size of layout information must be same as the size of number of "
800  "elements");
801  layoutInfoList = allocator.copyInto(keyLayoutInfo).data();
802  }
803 
804  const StructType::MemberDecorationInfo *memberDecorationList = nullptr;
805  unsigned numMemberDecorations = 0;
806  if (!std::get<2>(key).empty()) {
807  auto keyMemberDecorations = std::get<2>(key);
808  numMemberDecorations = keyMemberDecorations.size();
809  memberDecorationList = allocator.copyInto(keyMemberDecorations).data();
810  }
811  return new (allocator.allocate<StructTypeStorage>())
812  StructTypeStorage(keyTypes.size(), typesList, layoutInfoList,
813  numMemberDecorations, memberDecorationList);
814  }

816  ArrayRef<Type> getMemberTypes() const {
817  return ArrayRef<Type>(memberTypes, getSubclassData());
818  }
819 
820  ArrayRef<StructType::LayoutInfo> getLayoutInfo() const {
821  if (layoutInfo) {
822  return ArrayRef<StructType::LayoutInfo>(layoutInfo, getSubclassData());
823  }
824  return {};
825  }
826 
827  ArrayRef<StructType::MemberDecorationInfo> getMemberDecorationsInfo() const {
828  if (memberDecorationsInfo) {
829  return ArrayRef<StructType::MemberDecorationInfo>(memberDecorationsInfo,
830  numMemberDecorations);
831  }
832  return {};
833  }
834 
835  Type const *memberTypes;
836  StructType::LayoutInfo const *layoutInfo;
837  unsigned numMemberDecorations;
838  StructType::MemberDecorationInfo const *memberDecorationsInfo;
