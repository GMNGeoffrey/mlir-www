81 /// Returns the linearized list of all view dimensions in a linalgOp. Applying
82 /// the inverse, concatenated loopToOperandRangeMaps to this list allows the
83 /// derivation of loop ranges for any linalgOp.
84 template <typename ConcreteOp>
85 SmallVector<Value, 8> getViewSizes(OpBuilder &builder, ConcreteOp linalgOp) {
86  auto loc = linalgOp.getLoc();
87  SmallVector<Value, 8> res;
88  for (auto v : linalgOp.getInputsAndOutputBuffers()) {
89  MemRefType t = v.getType().template cast<MemRefType>();
90  for (unsigned i = 0; i < t.getRank(); ++i)
91  res.push_back(builder.create<DimOp>(loc, v, i));
92  }
93  return res;
94 }

131 /// Apply the permutation defined by `permutation` to `inVec`.
132 /// Element `i` in `inVec` is mapped to location `j = permutation[i]`.
133 /// E.g.: for an input vector `inVec = ['a', 'b', 'c']` and a permutation vector
134 /// `permutation = [2, 0, 1]`, this function leaves `inVec = ['c', 'a', 'b']`.
135 template <typename T, unsigned N>
136 void applyPermutationToVector(SmallVector<T, N> &inVec,
137  ArrayRef<unsigned> permutation) {
138  SmallVector<T, N> auxVec(inVec.size());
139  for (unsigned i = 0; i < permutation.size(); ++i)
140  auxVec[i] = inVec[permutation[i]];
141  inVec = auxVec;
142 }
