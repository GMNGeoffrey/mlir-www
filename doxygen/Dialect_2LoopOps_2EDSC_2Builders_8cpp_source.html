26 void mlir::edsc::ParallelLoopNestBuilder::operator()(
27  function_ref<void(void)> fun) {
28  if (fun)
29  fun();
30  // Iterate on the calling operator() on all the loops in the nest.
31  // The iteration order is from innermost to outermost because enter/exit needs
32  // to be asymmetric (i.e. enter() occurs on LoopBuilder construction, exit()
33  // occurs on calling operator()). The asymmetry is required for properly
34  // nesting imperfectly nested regions (see LoopBuilder::operator()).
35  for (auto lit = loops.rbegin(), eit = loops.rend(); lit != eit; ++lit)
36  (*lit)();
37 }

39 mlir::edsc::LoopNestBuilder::LoopNestBuilder(MutableArrayRef<Value> ivs,
40  ArrayRef<Value> lbs,
41  ArrayRef<Value> ubs,
42  ArrayRef<Value> steps) {
43  assert(ivs.size() == lbs.size() && "expected size of ivs and lbs to match");

45  assert(ivs.size() == steps.size() &&
46  "expected size of ivs and steps to match");
47  loops.reserve(ivs.size());
48  for (auto it : llvm::zip(ivs, lbs, ubs, steps))

50  std::get<2>(it), std::get<3>(it)));
51  assert(loops.size() == ivs.size() && "Mismatch loops vs ivs size");
52 }

81 LoopBuilder mlir::edsc::makeParallelLoopBuilder(MutableArrayRef<Value> ivs,
82  ArrayRef<Value> lbs,
83  ArrayRef<Value> ubs,
84  ArrayRef<Value> steps) {
85  loop::ParallelOp parallelOp = OperationBuilder<loop::ParallelOp>(
86  SmallVector<Value, 4>(lbs.begin(), lbs.end()),
87  SmallVector<Value, 4>(ubs.begin(), ubs.end()),
88  SmallVector<Value, 4>(steps.begin(), steps.end()));
89  for (size_t i = 0, e = ivs.size(); i < e; ++i)
90  ivs[i] = parallelOp.getBody()->getArgument(i);
91  LoopBuilder result;
92  result.enter(parallelOp.getBody());
93  return result;
94 }

96 mlir::edsc::LoopBuilder
97 mlir::edsc::makeLoopBuilder(Value *iv, Value lb, Value ub, Value step,
98  MutableArrayRef<Value> iterArgsHandles,
99  ValueRange iterArgsInitValues) {
100  mlir::edsc::LoopBuilder result;
101  loop::ForOp forOp =
102  OperationBuilder<loop::ForOp>(lb, ub, step, iterArgsInitValues);
103  *iv = Value(forOp.getInductionVar());
104  auto *body = loop::getForInductionVarOwner(*iv).getBody();
105  for (size_t i = 0, e = iterArgsHandles.size(); i < e; ++i) {
106  // Skipping the induction variable.
107  iterArgsHandles[i] = body->getArgument(i + 1);
108  }
109  result.setOp(forOp);
110  result.enter(body);
111  return result;
112 }
